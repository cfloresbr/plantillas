Ejemplo de Uso

A continuación, se muestra un ejemplo de cómo crear una guía simple usando ShepherdJS:

<button type="button" id="miBoton">< class="btn btn-primario">< Haz clic aquí</button>
Como incluirlo
<link rel="stylesheet" href="vendor/shepherd/dist/css/shepherd.css"/> <script src="vendor/shepherd/dist/js/shepherd.js"></script>
Ejemplo codigo 
                                                
                                        
                                                    
                                                    <script>
                                                    // Función para crear y ejecutar la guía
                                                    function crearGuia() {
                                                        // Crea una instancia de Shepherd
                                                        const tour = new Shepherd.Tour({
                                                            defaultStepOptions: {
                                                                classes: 'shepherd-theme-arrows',
                                                            },
                                                        });
                                            
                                                        // Agrega los pasos
                                                        tour.addStep({
                                                            id: 'paso-1',
                                                            text: 'Haz clic en este botón para comenzar.',
                                                            attachTo: {
                                                                element: '#miBoton',
                                                                on: 'bottom',
                                                            },
                                                            buttons: [
                                                                {
                                                                    text: 'Siguiente',
                                                                    action: tour.next,
                                                                },
                                                            ],
                                                        });
                                            
                                                        // Inicia la guía
                                                        tour.start();
                                                    }
                                            
                                                    // Ejecuta la guía cuando se carga el documento
                                                    document.addEventListener('DOMContentLoaded', function () {
                                                        crearGuia();
                                                    });
                                            
                                                    // Ejecuta la guía cuando se hace clic en el botón
                                                    document.getElementById('miBoton').addEventListener('click', function () {
                                                        crearGuia();
                                                    });
                                                    </script>
                                                
                                            
                                                
                                                    
                                                    <script>
                                                    function addStepIfElementExists(stepsArray, tour) {
                                                        const steps = [];
                                                    
                                                        // Iteramos a través de cada objeto en el array stepsArray
                                                        stepsArray.forEach((stepObj, index) => {
                                                            // Verificamos si el elemento existe, es visible y no está oculto
                                                            if ($(stepObj.elemento).length > 0 && $(stepObj.elemento).is(":visible") && !$(stepObj.elemento).is(":hidden")) {
                                                                // Creamos un objeto 'step' para el paso actual
                                                                const step = {
                                                                    classes: stepObj.classes,
                                                                    title: stepObj.title,
                                                                    text: stepObj.texto,
                                                                    attachTo: {
                                                                        element: stepObj.elemento,
                                                                        on: stepObj.on !== undefined ? stepObj.on : 'auto',
                                                                    },
                                                                    buttons: [],
                                                                    id: stepObj.id,
                                                                };
                                                    
                                                                // Agregamos un botón "Anterior" para todos los pasos excepto el primero
                                                                if (index > 0) {
                                                                    step.buttons.push({
                                                                        action: () => tour.back(), // Usamos el objeto 'tour' para llamar a los métodos
                                                                        secondary: true,
                                                                        text: 'Anterior',
                                                                    });
                                                                }
                                                    
                                                                // Agregamos un botón "Siguiente" para todos los pasos excepto el último
                                                                if (index < stepsArray.length - 1) {
                                                                    step.buttons.push({
                                                                        action: () => tour.next(), // Usamos el objeto 'tour' para llamar a los métodos
                                                                        text: 'Siguiente',
                                                                    });
                                                                }
                                                    
                                                                // Agregamos un botón "Hecho" para el último paso
                                                                if (index === stepsArray.length - 1) {
                                                                    step.buttons.push({
                                                                        action: () => tour.cancel(), // Usamos el objeto 'tour' para llamar a los métodos
                                                                        text: 'Hecho',
                                                                    });
                                                                }
                                                    
                                                                // Agregamos el paso al array de pasos
                                                                steps.push(step);
                                                            }
                                                        });
                                                    
                                                        // Devolvemos el array de pasos
                                                        return steps;
                                                    }

                                                            // Habilitar una superposición modal oscura de fondo
                                                            useModalOverlay: true

                                                        // Definir los pasos del tour en un array
                                                        let stepsArray = [
                                                            {
                                                                // Título del paso
                                                                title: "¿Qué es ShepherdJS?",

Puedes utilizar ShepherdJS para guiar a los usuarios a través de características o procesos importantes de tu sitio web.",
                                                                // Posición del resaltado en relación con el elemento (en la parte inferior)

                                                                // Posición del resaltado en relación con el elemento (en la parte inferior)
                                                                on: "bottom",
                                                                // Identificador único del paso
                                                                id: "#contenedorFunciona",
                                                                // Clases CSS adicionales para personalizar el aspecto del paso
                                                                classes: '',

                                                                // Título del paso
                                                                title: "Ejemplo de Uso",
                                                                // Elemento objetivo que se resaltará en la página
                                                                elemento: '#EjemploUso',
                                                                // Texto descriptivo del paso
                                                                texto: "A continuación, se muestra un ejemplo de cómo crear una guía simple usando ShepherdJS:",
                                                                // Posición del resaltado en relación con el elemento (en la parte inferior)
                                                                on: "bottom",
                                                                // Identificador único del paso
                                                                id: "#EjemploUso",
                                                                // Clases CSS adicionales para personalizar el aspecto del paso
                                                                classes: '',
                                                            },
                                                        ];
                                                    
                                                        // Utilizar la función addStepIfElementExists para verificar si los elementos objetivo existen en la página y crear los pasos correspondientes
                                                        const steps = addStepIfElementExists(stepsArray, tour);
                                                    
                                                        // Agregar los pasos al tour
                                                        tour.addSteps(steps);
                                                    
                                                        // Iniciar el tour
                                                        tour.start();
